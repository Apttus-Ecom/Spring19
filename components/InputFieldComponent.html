Description

Input field component

The Input field component displays the field values as a radio button, checkbox, dropdown, a text area or a label value, depending upon the field type.

Example Usage:

options
getoptions()

Adds the field values to be displayed, to an object with key value properties. The field would be a read only if the disabled property is set to true.

Returns : Array<PicklistValue> 
<div [ngSwitch]="field?.type" class="form-group">
    <label *ngIf="label" class="font-weight-bold">{{field?.label}}</label>
  
    <!--Picklist-->
    <div *ngSwitchCase="'picklist'">
      <select class="custom-select" [class.custom-select-sm]="small" *ngIf="field.picklistValues.length > 4; else checkboxes" [(ngModel)]="value" (change)="propagateChange(value)">
        <option *ngFor="let option of options" [value]="option.value" [selected]="option.value === value">
          {{option.label}}
        </option>
      </select>
      <ng-template #checkboxes>
        <div *ngFor="let option of options" class="custom-control custom-radio" [class.custom-radio-sm]="small" [class.custom-control-inline]="inline">
          <input type="radio" class="custom-control-input" [name]="uuid + '_' + field.name" [value]="option.value" [id]="uuid + '_' + field.name + '_' + option.value"
            [(ngModel)]="value" (change)="onRadioChange()" [disabled]="readonly || option.disabled">
          <label class="custom-control-label mr-auto" [for]="uuid + '_' + field.name + '_' + option.value">
            {{option.label}}
          </label>
        </div>
      </ng-template>
    </div>
  
    <!--Checkbox-->
    <select class="form-control" [class.form-control-sm]="small" *ngSwitchCase="'boolean'" [(ngModel)]="value" [class.custom-control-inline]="inline" (change)="propagateChange(value)">
      <option [value]="true">Yes</option>
      <option [value]="false">No</option>
    </select>
  
    <!--Multi select picklist-->
    <ng-container *ngSwitchCase="'multipicklist'">
      <div *ngFor="let option of field.picklistValues" class="custom-control custom-checkbox" [class.custom-checkbox-sm]="small" [class.custom-control-inline]="inline">
        <input type="checkbox" class="custom-control-input" [checked]="value?.indexOf(option.value) > -1" [id]="uuid + '_' + field.name + '_' + option.value" 
              (change)="onMultiChange(option.value)">
        <label class="custom-control-label" [for]="uuid + '_' + field.name + '_' + option.value">{{option.label}}</label>
      </div>
    </ng-container>
  
    <!-- String -->
    <input type="text" class="form-control" [class.form-control-sm]="small" [id]="value" [(ngModel)]="value" *ngSwitchCase="'text'" (change)="propagateChange(value)"/>
  
    <input type="number" class="form-control" [class.form-control-sm]="small" [id]="value" [(ngModel)]="value" *ngSwitchCase="'double'" (change)="propagateChange(value)"/>
  
    <div class="px-3" *ngSwitchCase="'string'">
      <textarea [(ngModel)]="value" class="form-control" [class.form-control-sm]="small" (change)="propagateChange(value)"></textarea>
    </div>

    <!-- Date -->
    <div class="form-group" *ngSwitchCase="'date'">
      <input type="text" [id]="uuid" class="form-control form-control-sm" bsDatepicker
        [(ngModel)]="value" [bsConfig]="bsConfig" (ngModelChange)="onDateChange()">
    </div>

    <!-- Percentage -->
    <div class="form-group" *ngSwitchCase="'percent'">
      <input type="text" [id]="uuid" class="form-control form-control-sm" [ngModel]="value/100 | percent:'1.2'" (blur)= "onPercentageChange($event)">
    </div>

    <!-- Currency -->
    <div class="form-group" *ngSwitchCase="'currency'">
      <input type="text" [id]="uuid" class="form-control form-control-sm" [ngModel]="value | currency: 'USD' : 'symbol': '1.2'" (blur)= "onCurrencyChange($event)">
    </div>

    <!-- Datetime -->
    <div class="form-group" *ngSwitchCase="'datetime'">
      <input type="text" [id]="uuid" class="form-control form-control-sm" bsDatepicker
        [(ngModel)]="value" [bsConfig]="bsConfig" (ngModelChange)="onDateChange()">
    </div>

  </div>
